def format(source_code, source_type):
    from pygments import highlight
    from pygments.lexers import get_lexer_by_name
    from pygments.styles import get_style_by_name
    from pygments.formatters import HtmlFormatter

    # https://pygments.org/styles/
    # ìœ„ ì‚¬ì´íŠ¸ë¥¼ ì°¸ê³ í•˜ì—¬ ì›í•˜ëŠ” ìŠ¤íƒ€ì¼ì„ ê³ ë¥¸ë‹¤
    style = get_style_by_name("nord")
    background_color = style.background_color
    lexer = get_lexer_by_name(source_type, stripall=True)
    formatter = HtmlFormatter(
        style=style,
        linenos="table",
        cssclass="code",
        wrapcode=True,
    )

    css = formatter.get_style_defs()
    result = highlight(source_code, lexer, formatter)
    import re
    css = re.sub(re.compile("/\*.*?\*/", re.DOTALL), "", css)
    css = "".join([x.strip() for x in css.split("\n")])
    html = f"""
    <div style="
            background-color: {background_color};
            padding-top: 1.25rem;
            padding-bottom: 1.25rem;
        ">
        <style scoped>{css}</style>
        @HTMLCODE
    </div>
    """
    html = " ".join([x.strip() for x in html.strip().split("\n")])
    html = html.replace("@HTMLCODE", result)
    return html
